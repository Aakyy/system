package hibernatejava; 
import org.hibernate.Session; 
import org.hibernate.SessionFactory; 
import org.hibernate.Transaction; 
import org.hibernate.cfg.Configuration; 
import java.util.ArrayList; 
import java.util.List; 
public class AddEmployee { 
    public static void main(String[] args) { 
        Configuration config = new Configuration().configure().addAnnotatedClass(Employee.class); 
        SessionFactory factory = config.buildSessionFactory(); 
        Session session = factory.openSession(); 
        Transaction tx = session.beginTransaction(); 
        List<Employee> employees = createEmployeeList(); 
        for (Employee employee : employees) { 
            session.save(employee); 
        } 
        tx.commit(); 
        session.close(); 
        factory.close(); 
        System.out.println("Data entered successfully!"); 
    }

2ND Query: Write an Hibernate API to update the name of employee on id 1 
UpdateEmployeeName.java 
package hibernatejava; 
import org.hibernate.Session; 
import org.hibernate.SessionFactory; 
import org.hibernate.Transaction; 
import org.hibernate.cfg.Configuration; 
 
public class UpdateEmployeeName { 
    public static void main(String[] args) { 
        SessionFactory factory = new Configuration().configure().addAnnotatedClass(Employee.class).buildSessionFactory(); 
        Session session = factory.openSession(); 
        Transaction tx = null; 
 
        try { 
            tx = session.beginTransaction(); 
            Employee employeeToUpdate = session.get(Employee.class, 1); 
            if (employeeToUpdate != null) { 
                employeeToUpdate.setName("Yash Parmar"); 
            } 
            tx.commit(); 
            System.out.println("Name Updated successfully!"); 
        } catch (Exception e) { 
            if (tx != null && tx.isActive()) { 
                tx.rollback(); 
            } 
            e.printStackTrace(); 
        } finally { 
            session.close(); 
            factory.close(); 
        } 
    } 
} 

3RD Query: Write an Hibernate API to delete an employee on id 3 
DeleteEmployee.java 
package hibernatejava; 
import org.hibernate.Session; 
import org.hibernate.SessionFactory; 
 
import org.hibernate.Transaction; 
import org.hibernate.cfg.Configuration; 
public class DeleteEmployee { 
    public static void main(String[] args) { 
        SessionFactory factory = new Configuration().configure().addAnnotatedClass(Employee.class).buildSessionFactory(); 
        Session session = factory.openSession(); 
        Transaction tx = session.beginTransaction(); 
        Employee employeeToDelete = session.get(Employee.class, 3); 
        if (employeeToDelete != null) { 
            session.delete(employeeToDelete); 
        } 
 
        tx.commit(); 
        session.close(); 
        factory.close(); 
        System.out.println("Employee on id 3 successfully Deleted!"); 
    } 
}

package mypackage2; 
import java.io.IOException; 
import java.io.PrintWriter; 
import jakarta.servlet.ServletException; 
import jakarta.servlet.annotation.WebServlet; 
import jakarta.servlet.http.HttpServlet; 
import jakarta.servlet.http.HttpServletRequest; 
import jakarta.servlet.http.HttpServletResponse; 
@WebServlet("/Lengthofthename") 
public class Lengthofthename extends HttpServlet { 
private static final long serialVersionUID = 1L; 
protected void doPost(HttpServletRequest request, HttpServletResponse response) 
throws ServletException, IOException { 
// Get the user-entered name parameter from the form 
String userName = request.getParameter("userName"); 
// Calculate the length of the name 
int nameLength = userName.length(); 
// Set the response content type 
response.setContentType("text/html"); 
// Get the PrintWriter 
PrintWriter out = response.getWriter(); 
// Write the HTML response 
out.println("<html>"); 
out.println("<head><title>Name Length</title></head>"); 
out.println("<body>"); 
out.println("<h2>Hello, " + userName + "!</h2>"); 
out.println("<p>Your name has " + nameLength + " characters.</p>"); 
out.println("</body>"); 
out.println("</html>"); 
} 
}
